class Solution {
public:
    bool isAnagram(string s, string t) {
        unordered_map<char,int> umap;
        if(s.size() != t.size()) return 0;
        for(int i=0;i<s.size();i++){
            umap[s[i]]++;
            umap[t[i]]--; 
        }
        for(auto i:umap)
            if(i.second != 0) return 0;
        return 1;
    }
};
